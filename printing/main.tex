\subsection*{Example 1}
\begin{minted}[fontsize=\Large]{python}
import math

def f(u, v):
    l = len(u)
    s = 0
    for i in range(l):
        s += (u[i] - v[i]) ** 2
    return math.sqrt(s)
\end{minted}

\subsection*{Example 2}
\begin{minted}[fontsize=\Large]{python}
import math 

def f(u, v, isM, isE):
    if isM == 1:
        l = len(u)
        s = 0
        for i in range(l):
            s += abs(u[i] - v[i])
        return s
    
    if isE == 1:
        l = len(u)
        s = 0
        for i in range(l):
            s += (u[i] - v[i]) ** 2
        return math.sqrt(s)
\end{minted}
